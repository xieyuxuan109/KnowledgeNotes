## 全局中间件
### middwire下面创建global.go写全局中间件
```
type bodyCopy struct {
	http.ResponseWriter               //结构体嵌入接口类型
	body                *bytes.Buffer //记录响应体
}
func NewBodyCopy(w http.ResponseWriter) *bodyCopy {
	return &bodyCopy{
		ResponseWriter: w,
		body:           bytes.NewBuffer([]byte{}),
	}
}
func (bc bodyCopy) Write(b []byte) (int, error) {
	//1.先记录响应内容
	bc.body.Write(b)
	//2.再往http响应写报告内容
	return bc.ResponseWriter.Write(b)
}
// 定义全局中间件
// rest.Middleware->type Middleware func(next http.HandlerFunc) http.HandlerFunc
// 复制请求响应体
// type HandlerFunc func(ResponseWriter, *Request)
func CopyResp(next http.HandlerFunc) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		// 创建 BodyCopy 包装器
		bc := NewBodyCopy(w)
		// 只执行一次业务逻辑，响应写入 bc
		next(bc, r)
		// 打印捕获的响应体
		fmt.Printf("req:%v resp:%v\n", r.URL, bc.body.String())
		// bc 已经自动写入了原始的 w，不需要再调用 next
		// 注意：BodyCopy 应该在 Write 时同步写入原始 ResponseWriter
	}
}
func MiddlewareWithAnotherService(ok bool) rest.Middleware {
	return func(next http.HandlerFunc) http.HandlerFunc {
		return func(w http.ResponseWriter, r *http.Request) {
			if ok {
				fmt.Println("ok")
			}
			next(w, r)
		}
	}
}
```
### 在user.go中添加中间件
```
server.Use(middleware.CopyResp)
server.Use(middleware.MiddlewareWithAnotherService(true))
```
